   def speak(self):

2)Create a class Book with title and author, and a method to display them

class Book:
   def __init__(self,author,title):
      self.author=author
      self.title=title
   def display(self):

4)Create a class Rectangle with length and width, and method area().

class Rectangle:
   def __init__(self,length,width):
      self.length=length
      self.width=width
   def area(self):
      return self.length*self.width
r=Rectangle(6,7)
print("Area:",r.area())
